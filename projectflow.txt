    mongo:
        user: 
        password: 
        host: 
        cluster: 
--> load yaml and use urllib.url_parse to encode special characters in usename and password to avoid conflict in --mongodb connection string-- special characters

import yaml
from urllib.parse import quote_plus
#quote plus is used to encode special characters in username and password, 
#if username and password contains special characters they are escaped so that there is no contradiction
#with ? and @ used in mongodd connection string

config = yaml.safe_load(open(r"../project_cred.yaml"))

user = quote_plus(config["mongo"]["user"])
password = quote_plus(config["mongo"]["password"])
host = config["mongo"]["host"]
cluster = config["mongo"]["cluster"]
